(defmacro success->
  "When (success? expr) is truthy, threads it into the first form (via ->),
  and when (success? that result) is truthy, through the next etc"
  [expr & forms]
  (let [g (gensym)
        pstep (fn [step] `(if (success? ~g) (-> ~g ~step) ~g))]
    `(let [~g ~expr
           ~@(interleave (repeat g) (map pstep forms))]
       ~g)))
